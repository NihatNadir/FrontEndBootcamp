Fonksiyon tanımlama

            function sum(x,y){
                let s = x+y;
                return s;
            }
            let result = sum(4,5);
            console.log(result);

Anonim Fonksiyon tanımlama

            let carp = function(x,y) {
                return x * y;
            }
            console.log(carp(4, 5));

            let cikart = (x,y) => x-y;
            
            console.log(cikart(10, 2));
